# 滚动判断

> 滚动时触发handleScroll方法

```vue
		<view class="detail">
			<view class="detailContrainer" ref="detailContainer" @scroll="handleScroll">
				<view class="detailItem" v-for="(item, index) in rechargeInformation" :key="index">

					<view class="icon">
						<u-icon :name="dealWithIcon(item.depositMethod)" :color="dealWithColor(item.depositMethod)"
							size="50" @click="closeNotice"></u-icon>
					</view>

					<view class="rightContrainer">
						<view class="detailName">
							<view class="name">{{ dealWithDepositMethod(item.depositMethod) }}</view>
							<view class="priceInOut" :style="{ color: item.flowType === 0 ? 'red' : '' }">{{ item.flowType==0?"-":"+" }}{{ item.amount.toFixed(2) }} </view>
						</view>
						<view class="price">
							<view class="time">{{ dealWithDateTime(item.flowDate) }}</view>
							<view class="balance"> 零钱余额 {{ item.balance?item.balance.toFixed(2):0 }} </view>
						</view>
					</view>
				</view>
			</view>
		</view>


			handleScroll(event) {
				const container = this.$refs.detailContainer;
				const scrollTop = container.$el.scrollTop;
				const scrollHeight = container.$el.scrollHeight;
				const clientHeight = container.$el.clientHeight;
				if (scrollTop + clientHeight >= scrollHeight) {
					// 判断是否已经获取了所有数据
					if (this.dto.current < this.totalCurrent) {
						// 创建新的副本，给current加1
						this.isCurrent = true
						this.dto.current = this.dto.current + 1
						// 等待请求结束
						this.getSelectList(this.dto)
						this.isCurrent = false
					}
				}
			},
```
